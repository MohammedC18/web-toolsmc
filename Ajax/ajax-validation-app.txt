below the submit button.

mkdir ajax-validation

cd ajax-validation


// ======================================
// STEP 2
// Create File
// ======================================

New-Item index.html -ItemType File


// ======================================
// STEP 3
// Open Project in VS Code
// ======================================

code .


// ======================================
// STEP 4
// Open:
// index.html
// ======================================

<!DOCTYPE html>
<html>

<head>

  <title>AJAX User Validation</title>

</head>

<body>

<h1>AJAX User Validation</h1>

<form>

  <!-- USERNAME -->

  <input
    type="text"
    id="username"
    placeholder="Enter Username"
  >

  <br><br>


  <!-- PASSWORD -->

  <input
    type="password"
    id="password"
    placeholder="Enter Password"
  >

  <br><br>


  <!-- SUBMIT BUTTON -->

  <button
    type="button"
    onclick="validateUser()"
  >
    Submit
  </button>

</form>

<br>


<!-- RESULT DISPLAY -->

<h2 id="result"></h2>


<script>

function validateUser()
{

  let username =
    document.getElementById("username").value;

  let password =
    document.getElementById("password").value;


  // AJAX SIMULATION

  setTimeout(function()
  {

    if(username == "admin" && password == "1234")
    {

      document.getElementById("result").innerHTML =
        "Valid User";

    }
    else
    {

      document.getElementById("result").innerHTML =
        "Invalid User";

    }

  }, 500);

}

</script>

</body>
</html>

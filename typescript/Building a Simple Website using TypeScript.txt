
mkdir simple-website

// Step 4: Write Code in app.ts

function showMessage(): void {
    const message = document.getElementById("msg");

    if (message) {
        message.innerHTML = "Welcome to TypeScript Website!";
    }
}
// Step 5: Write Code in index.html

<!DOCTYPE html>
<html>

<head>
    <title>TypeScript Website</title>
    <link rel="stylesheet" href="style.css">
</head>

<body>

    <div class="container">
        <h1>Simple TypeScript Website</h1>

        <button onclick="showMessage()">
            Click Me
        </button>

        <p id="msg"></p>
    </div>

    <script src="app.js"></script>

</body>
</html>
